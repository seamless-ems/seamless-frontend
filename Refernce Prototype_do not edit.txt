    /**
     * Switches between main app screens (Login, Dashboard, Event Detail, Speaker List, etc.)
     * DEV NOTE: Uses classList for screen management (preferred pattern)
     * @param {string} screenId - ID of screen to show (e.g., 'screen-login', 'screen-dashboard')
     */
    function showScreen(screenId) {
      // Hide all screens
      document.querySelectorAll('.screen').forEach(screen => {
        screen.classList.remove('active');
      });

      // Show target screen
      document.getElementById(screenId).classList.add('active');

      // Close any open menus
      document.querySelectorAll('.user-dropdown-menu, .event-card-menu-dropdown').forEach(menu => {
        menu.classList.remove('active');
      });

      // Scroll to top
      window.scrollTo(0, 0);
    }

    /**
     * Toggles user dropdown menu (top right)
     * DEV NOTE: Finds menu relative to clicked element to work across all screens
     * @param {Event} event - Click event from the toggle button
     */
    function toggleUserMenu(event) {
      // Find the menu within the same user-dropdown container
      const dropdown = event ? event.currentTarget.closest('.user-dropdown') : document.querySelector('.user-dropdown');
      const menu = dropdown.querySelector('.user-dropdown-menu');
      menu.classList.toggle('active');
    }

    /**
     * Toggles event card ⋮ menu (3-dot menu on event cards)
     * Closes other open event menus to maintain single-menu UX
     * DEV NOTE: Uses classList.toggle for consistency with toggleUserMenu
     * @param {string} menuId - ID of menu to toggle
     */
    function toggleEventMenu(menuId) {
      event.stopPropagation();
      const menu = document.getElementById(menuId);

      // Close other menus
      document.querySelectorAll('.event-card-menu-dropdown').forEach(m => {
        if (m.id !== menuId) m.classList.remove('active');
      });

      menu.classList.toggle('active');
    }

      document.getElementById('speaker-tab-forms').style.display = 'none';

      // Update tab button styles

      const applicationsTab = document.getElementById('tab-applications');
      const formsTab = document.getElementById('tab-forms');
      const embedBuilderTab = document.getElementById('tab-embed-builder');

      // Reset all tabs
      [speakersTab, applicationsTab, formsTab, embedBuilderTab].forEach(tab => {
        tab.style.color = 'var(--text-secondary)';
        tab.style.borderBottomColor = 'transparent';
      });

      // Highlight selected tab
      if (tabName === 'speakers') {
        speakersTab.style.color = 'var(--primary)';
        speakersTab.style.borderBottomColor = 'var(--primary)';

        embedBuilderTab.style.color = 'var(--primary)';

      }
    }

      // Update modal title

      document.getElementById('form-name-input').value = formName;

      const applicationLabel = document.getElementById('form-type-application-label');

        applicationLabel.style.background = 'var(--primary-subtle)';
        speakerInfoLabel.style.borderColor = 'var(--border-default)';
        speakerInfoLabel.style.background = 'var(--bg-surface)';
      }

      // Reset all checkboxes first
      document.getElementById('field-bio-checkbox').checked = false;
      document.getElementById('field-headshot-checkbox').checked = false;

      document.getElementById('form-save-button').textContent = 'Update Form';

      document.getElementById('modal-create-form').style.display = 'flex';

    /**
     * Click-outside-to-close behavior for dropdown menus
     * DEV NOTE: Uses event delegation on document to handle all menus
     */
    document.addEventListener('click', function(e) {
      // Close user dropdown and event card menus when clicking outside
      if (!e.target.closest('.user-dropdown') && !e.target.closest('.event-card-menu')) {
        document.querySelectorAll('.user-dropdown-menu, .event-card-menu-dropdown').forEach(menu => {
          menu.classList.remove('active');
        });
      }

      // Close speaker menu if clicking outside
      // DEV NOTE: Uses style.display (inconsistent with other menus - could be refactored)
      const speakerMenu = document.getElementById('speakerMenu');
      if (speakerMenu && !e.target.closest('#speakerMenu') && e.target.getAttribute('onclick') !== 'toggleSpeakerMenu(event)') {
        speakerMenu.style.display = 'none';
      }
    });

    /**
     * Switches between Account Dashboard tabs (Current Events / Past Events)
     * DEV NOTE: Uses classList approach (cleaner than showSpeakerTab's style.display approach)
     * Pattern: This is the preferred tab switching pattern for refactoring
     * @param {string} tabName - Tab identifier (e.g., 'current', 'past')
     */
    function switchTab(tabName) {
      // Update tab buttons
      document.querySelectorAll('.tab').forEach(tab => {
        tab.classList.remove('active');
      });
      event.target.classList.add('active');

      // Update tab content
      document.querySelectorAll('.tab-content').forEach(content => {
        content.classList.remove('active');
      });
      document.getElementById('tab-' + tabName).classList.add('active');
    }
